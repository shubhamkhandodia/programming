class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        
        vector<int> hashtable;
        vector<int> indices;
        
        int element,index1,index2,i=0;
        bool present = false;
        
        while(present!=true)
        {
            element = target-nums[i];
            
            std::vector<int>::iterator it = std::find(hashtable.begin(), hashtable.end(), element);
            
            if (it != hashtable.end())
            {
                present=true;
                index1 = std::distance(hashtable.begin(), it);
                index2 = i;
                indices.push_back(index1);
                indices.push_back(index2);
            }
                
            else
            {
                hashtable.push_back(nums[i]);
                i++;
            }
            
        }
        
        return indices;
        
    }
};
